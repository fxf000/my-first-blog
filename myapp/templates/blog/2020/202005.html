I wrote an elementary program about Natural Language Processing (NLP). The program is called N-gram and is well known program in NLP. N-gram is N consecutive words or letters, And in this time, I wrote an N-gram program with two cases of words and letters.

N-gram (05: NLP100 Exercise) 


            def N_gram_word(sentence,N):
                words = sentence.replace(".","")
                word = words.split()
                N_gram_word_list = []
                if N > len(word):
                    return word
                else:
                    for i in range(len(word)-(N-1)):
                        N_gram_word_list.append(word[i:i+N])
                    return N_gram_word_list

            def N_gram_letter(sentence,N):
                words = sentence.replace(".","")
                word = words.split()
                letters = []
                for i in range(len(word)):
                    for j in range(len(word[i])):
                        letters.append(word[i][j])
                if N > len(letters):
                    return letters
                else:
                    N_gram_letter = []
                    for i in range(len(letters)-(N-1)):
                        N_gram_letter.append(letters[i:i+N])
                    return N_gram_letter

            sentence = "I am an NLPer."

            print("word bi-grams: ",N_gram_word(sentence,2))
            print("letter bi-grams: ",N_gram_letter(sentence,2))

Set (06: NLP100 Exercise) 


            def N_gram_letter(sentence,N):
                words = sentence.replace(".","")
                word = words.split()
                letters = []
                for i in range(len(word)):
                    for j in range(len(word[i])):
                        letters.append(word[i][j])
                if N > len(letters):
                    return letters
                else:
                    N_gram_letter = []
                    for i in range(len(letters)-(N-1)):
                        N_gram_letter.append(letters[i:i+N])
                    return N_gram_letter

            word1 = "paraparaparadise"
            word2 = "paragraph"

            X = N_gram_letter(word1,2)
            Y = N_gram_letter(word2,2)

            setX = set(map(tuple, X))
            setY = set(map(tuple, Y))

            unionXY = setX | setY
            intersectionXY = setX & setY
            differenceXY = setX - setY

            se = set(map(tuple, N_gram_letter("se",2)))

            print("X ∪ Y = {}".format(unionXY))
            print("X ∩ Y = {}".format(intersectionXY))
            print("X ∖ Y = {}".format(differenceXY))
            print("bi-gram se -> setX:{}, setY:{}".format(se.issubset(setX),se.issubset(setY)))

Maximum Profit (ALDS1_1_D: Aizu Online Judge) 


            n = int(input())
            R = []

            for i in range(n):
                r = int(input())
                R.append(r)

            minv = R[0]
            max_diff = -2000000000

            for j in range(1,n,1):

I wrote two programs about Natural Language Processing. The first program encrypts and decrypts English messages, and the second program randomly changes the order of letters other than the first and last.

Cipher text (08: NLP100 Exercise) 


            def cipher(message):
                message_list = []
                result = ""
                for i in range(len(message)):
                    if message[i].islower():
                        message_list.insert(i,chr(219-ord(message[i])))
                    else:
                        message_list.insert(i,message[i])
                for j in range(len(message)):
                    result += message_list[j]

                return result

            message = "I've seen things you people wouldn't believe."

            encription = cipher(message)
            decription = cipher(encription)

            print("encription: \n{}".format(encription))
            print("decription: \n{}".format(decription))

Typoglycemia (09: NLP100 Exercise) 


            import random

            def Typoglycemia(sentence):
                result = []
                for word in sentence.split(" "):
                    if len(word) <= 4:
                        result.append(word)
                    else:
                        shuffle_word_list = list(word[1:-1])
                        random.shuffle(shuffle_word_list)
                        typoglycemia_list = word[0]+"".join(shuffle_word_list)+word[-1]
                        result.append(typoglycemia_list)
                return " ".join(result)

            sentence = "I couldn’t believe that I could actually understand what I was reading
            : the phenomenal power of the human mind ."

            print(Typoglycemia(sentence))

I wrote a program which is called Insertion Sort. The algorithm sorts a sequence A in ascending order. The time complexity is O(kN) when each element in the input is position k away from its sorted position.

Insertion Sort (ALDS1_1_A: Aizu Online Judge) 


            def InsertionSort(S,N):
                for j in range(1,N,1):
                    v = S[j]
                    k = j-1
                    while k >= 0 and S[k] > v:
                        S[k+1] = S[k]
                        k -= 1
                    S[k+1] = v
                    print(" ".join(map(str,S)))

            N = int(input())
            S = [int(x) for x in input().split()]

            print(" ".join(map(str,S)))
            InsertionSort(S,N)
