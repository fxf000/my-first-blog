Zipf’s law (39: NLP100 Exercise) 


          from collections import defaultdict
          import matplotlib.pyplot as plt
          import japanize_matplotlib

          file = "./neko.txt.mecab"

          sentence = []
          sentences = []

          with open(file=file,mode="r") as f:
              for line in f:
                  if line != "EOS\n":
                      fields = line.split("\t")
                      if len(fields) != 2 or fields[0] == "":
                          continue
                      else:
                          attribute = fields[1].split(",")
                          if attribute[1] != "空白":
                              morpheme = {"surface":fields[0],
                                          "base":attribute[6],
                                          "pos":attribute[0],
                                          "pos1":attribute[1]}
                              sentence.append(morpheme)
                  else:
                      sentences.append(sentence)
                      sentence = []

          word_frequency = defaultdict(int)

          for sentence in sentences:
              for morpheme in sentence:
                  if morpheme["pos"] != "記号" and morpheme["base"] != "*\n":
                      word_frequency[morpheme["base"]] += 1

          ans = sorted(word_frequency.items(),key=lambda x:x[1],reverse=True)
          frequency_count = defaultdict(int)

          for a in ans:
              frequency_count[a[1]] += 1

          X = [x for x in range(1,len(ans)+1,1)]
          Y = [a[1] for a in ans]

          plt.figure(figsize=(8, 4))
          plt.scatter(X,Y,c="k",s=10)
          plt.xscale('log')
          plt.yscale('log')
          plt.xlabel("出現頻度の順位")
          plt.ylabel("出現頻度")
          plt.title("Zipfの法則")

          plt.show()

          import sys


          class Node():
              def __init__(self,key):
                  self.key = key
                  self.parent = None
                  self.left = None
                  self.right = None


          class BinaryTree():
              def __init__(self):
                  self.root = None

              def insert(self,key):
                  y = None
                  x = self.root
                  z = Node(key)
                  while x != None:
                      y = x
                      if z.key < x.key:
                          x = x.left
                      else:
                          x = x.right
                  z.parent = y
                  if y == None:
                      self.root = z
                  else:
                      if z.key < y.key:
                          y.left = z
                      else:
                          y.right = z

              def find(self,key):
                  v = Node(key)
                  w = self.root
                  while w != None and v.key != w.key:
                      if v.key < w.key:
                          w = w.left
                      else:
                          w = w.right
                  return w

              def delete(self,key):
                  z = self.find(key)
                  try:
                      if z.left == None or z.right == None:
                          y = z
                      else:
                          y = self.getSuccessor(z)
                          print(y.key)
                      if y.left != None:
                          x = y.left
                      else:
                          x = y.right
                      if x != None:
                          x.parent = y.parent
                      if y.parent == None:
                          self.root = x
                      elif y == y.parent.left:
                          y.parent.left = x
                      else:
                          y.parent.right = x
                      if y != z:
                          z.key = y.key
                  except AttributeError:
                      print("Not exist sucn a number...")

              def getSuccessor(self,u):
                  v = u.right
                  while v.left != None:
                      v = v.left
                  return v

              def inorder(self,u):
                  if u == None:
                      return
                  self.inorder(u.left)
                  print(" "+str(u.key),end="")
                  self.inorder(u.right)

              def preorder(self,u):
                  if u == None:
                      return
                  print(" "+str(u.key),end="")
                  self.preorder(u.left)
                  self.preorder(u.right)

              def output_order(self):
                  self.inorder(self.root)
                  print("")
                  self.preorder(self.root)
                  print("")


          def main():
              m = int(sys.stdin.readline())
              Tree = BinaryTree()

              for _ in range(m):
                  command = sys.stdin.readline().split()
                  if command[0] == "insert":
                      Tree.insert(int(command[1]))
                  elif command[0] == "find":
                      ans = Tree.find(int(command[1]))
                      if ans != None:
                          print("yes")
                      else:
                          print("no")
                  elif command[0] == "delete":
                      Tree.delete(int(command[1]))
                  else:
                      Tree.output_order()


          if __name__ == "__main__":
              main()

Maximum Heap (ALDS1_9_B: Aize Online Judge) 


          import sys

          def maxHeapify(A,H,i):
              l = 2*i
              r = 2*i+1
              if l <= H and A[l-1] > A[i-1]:
                  largest = l
              else:
                  largest = i
              if r <= H and A[r-1] > A[largest-1]:
                  largest = r
              if largest != i:
                  A[i-1],A[largest-1] = A[largest-1],A[i-1]
                  maxHeapify(A,H,largest)

          def main():
              H = int(sys.stdin.readline())
              A = list(map(int,sys.stdin.readline().split()))

              for i in range(H//2,0,-1):
                  maxHeapify(A,H,i)

              print(" ",end="")
              print(" ".join(map(str,A)))

Priority Queue (ALDS1_9_C: Aize Online Judge) 


          import sys

          max = 2000000
          A = [None]*(max+1)
          H = 0

          def insert(key):
              global H
              H += 1
              A[H-1] = -float("inf")
              heapIncreaseKey(A,H,key)

          def heapIncreaseKey(A,i,key):
              if key < A[i-1]:
                  return
              A[i-1] = key
              while i > 1 and A[i//2-1] < A[i-1]:
                  A[i-1],A[i//2-1] = A[i//2-1],A[i-1]
                  i = i//2

          def heapExtractMax(A):
              global H
              if H < 1:
                  return -float("inf")
              max = A[0]
              A[0] = A[H-1]
              H -= 1
              maxHeapify(A,1)
              return max

          def maxHeapify(A,i):
              global H
              l = 2*i
              r = 2*i+1
              if l <= H and A[l-1] > A[i-1]:
                  largest = l
              else:
                  largest = i
              if r <= H and A[r-1] > A[largest-1]:
                  largest = r
              if largest != i:
                  A[i-1],A[largest-1] = A[largest-1],A[i-1]
                  maxHeapify(A,largest)

          def main():
              for line in sys.stdin:
                  if line[0:6] == "insert":
                      insert(int(line[7:]))
                  elif line[0:7] == "extract":
                      print(heapExtractMax(A))
                  else:
                      break


          if __name__ == "__main__":
              main()

Longest Common Subsequence (ALDS1_10_C: Aize Online Judge) 


          import sys

          def LCS(X,Y):
              M = len(X)
              N = len(Y)
              C = [[None for _ in range(N+1)] for _ in range(M+1)]
              for i in range(0,M+1,1):
                  C[i][0] = 0
              for j in range(1,N+1,1):
                  C[0][j] = 0
              for i in range(1,M+1,1):
                  for j in range(1,N+1,1):
                      if X[i-1] == Y[j-1]:
                          C[i][j] = C[i-1][j-1] + 1
                      else:
                          C[i][j] = max(C[i][j-1],C[i-1][j])
              return C[M][N]

          def main():
              input = sys.stdin.readline
              q = int(input())

              for _ in range(q):
                  x = input().strip("\n")
                  y = input().strip("\n")
                  print(LCS(x,y))

Matrix Chain Multiplication (ALDS1_10_B: Aize Online Judge) 


          import sys


          def main():
              input = sys.stdin.readline
              n = int(input())

              m = [[None for _ in range(n)] for _ in range(n)]
              p = [None for _ in range(n+1)]

              def matrixChainMultiplication(m,p):
                  for i in range(1,n+1,1):
                      m[i-1][i-1] = 0
                  for l in range(2,n+1,1):
                      for i in range(1,n-l+2,1):
                          j = i+l-1
                          m[i-1][j-1] = float("inf")
                          for k in range(i,j,1):
                              m[i-1][j-1] = min(m[i-1][j-1],m[i-1][k-1]+m[k][j-1]+p[i-2]*p[k-1]*p[j-1])
                  return m[0][n-1]

              for i in range(n):
                  p[i],p[i+1] = map(int,input().split())

              print(matrixChainMultiplication(m,p))


          if __name__ == "__main__":
              main()
